# Load essential packages
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from google.colab import files
from statsmodels.tsa.stattools import adfuller, coint
from statsmodels.tsa.arima.model import ARIMA
import statsmodels.api as sm  # Explicitly import for OLS

# Function to load data
def load_ticker_data(ticker_name, date_range):
    print(f"Please upload a file with 5-min {ticker_name} data for {date_range} (earliest to latest):")
    uploaded = files.upload()
    file_key = next(iter(uploaded))
    return pd.read_excel(file_key)  # Switch to pd.read_csv for CSV files

# Function to get price series
def extract_prices(dataframe, ticker):
    print(f"\n{ticker} columns:", dataframe.columns.tolist())
    price_col = input(f"Select the closing price column for {ticker}: ")
    return dataframe[price_col].dropna().reset_index(drop=True)  # Reset index to ensure alignment

# Stationarity check
def check_stationarity(price_data, label):
    print(f"\nTesting {label} for stationarity:")
    adf_out = adfuller(price_data)
    stats = {'Statistic': adf_out[0], 'p-value': adf_out[1],
             'Critical 1%': adf_out[4]['1%'], 'Critical 5%': adf_out[4]['5%']}
    for key, value in stats.items():
        print(f"{key}: {value:.4f}")
    print("Stationary" if adf_out[1] < 0.05 else "Not Stationary")
    return adf_out[1]

# Cointegration analysis
def test_cointegration(series1, series2, name1, name2):
    print(f"\nChecking cointegration between {name1} and {name2}:")
    try:
        stat, pval, crit = coint(series1, series2)
        print(f"Statistic: {stat:.4f}, p-value: {pval:.4f}")
        print(f"Critical values: {crit}")
        if pval < 0.05:
            print("Cointegrated at 95% confidence")
        return stat, pval
    except Exception as e:
        print(f"Cointegration test failed: {e}")
        return None, None

# Fit regression for beta
def compute_beta(y_series, x_series):
    try:
        x_with_const = sm.add_constant(x_series)  # Add constant for intercept
        regression = sm.OLS(y_series, x_with_const).fit()
        return regression.params[1]  # Beta is the slope coefficient
    except Exception as e:
        print(f"Regression failed: {e}")
        return None

# ARIMA model selection
def optimize_arima(data, p_range=3, d_range=2, q_range=3):
    lowest_aic = float('inf')
    best_params = (0, 0, 0)
    for p in range(p_range):
        for d in range(d_range):
            for q in range(q_range):
                try:
                    model = ARIMA(data, order=(p, d, q)).fit()
                    if model.aic < lowest_aic:
                        lowest_aic = model.aic
                        best_params = (p, d, q)
                except:
                    pass
    print(f"\nBest ARIMA parameters: {best_params}, AIC: {lowest_aic:.2f}")
    return best_params

# Main execution
date_range = "March 4-5, 2025"
aal_df = load_ticker_data("AAL", date_range)
dal_df = load_ticker_data("DAL", date_range)

aal_series = extract_prices(aal_df, "AAL")
dal_series = extract_prices(dal_df, "DAL")

# Sync data lengths
if len(aal_series) != len(dal_series):
    synced_length = min(len(aal_series), len(dal_series))
    aal_series = aal_series[:synced_length]
    dal_series = dal_series[:synced_length]
    print(f"\nSynced to {synced_length} data points")
else:
    print(f"\nData points: {len(aal_series)}")

# Perform stationarity tests
check_stationarity(aal_series, "AAL")
check_stationarity(dal_series, "DAL")

# Cointegration and beta
coint_stat, coint_pval = test_cointegration(aal_series, dal_series, "AAL", "DAL")
if coint_pval is None:
    raise ValueError("Cointegration test failed, cannot proceed.")

beta_value = compute_beta(aal_series, dal_series)
if beta_value is None:
    raise ValueError("Beta computation failed, cannot proceed.")
print(f"Beta coefficient: {beta_value:.4f}")

# Spread calculation and test
spread_series = aal_series - beta_value * dal_series
check_stationarity(spread_series, "Spread")

# ARIMA fitting and forecasting
arima_params = optimize_arima(spread_series)
fitted_model = ARIMA(spread_series, order=arima_params).fit()
forecast_len = 60  # 5 hours at 5-min intervals
forecast_data = fitted_model.forecast(steps=forecast_len)
conf_bounds = fitted_model.get_forecast(steps=forecast_len).conf_int()

# Time setup for March 6, 2025
base_time = pd.to_datetime("2025-03-06 09:30:00")
time_points = [base_time + pd.Timedelta(minutes=5*i) for i in range(forecast_len)]
time_display = [t.strftime("%H:%M") for t in time_points]
forecast_range = list(range(len(spread_series), len(spread_series) + forecast_len))

# Visualization
plt.figure(figsize=(10, 6))
plt.plot(spread_series.index, spread_series, color='green', label='Past Spread')
plt.plot(forecast_range, forecast_data, color='purple', label='Forecasted Spread')
plt.fill_between(forecast_range, conf_bounds.iloc[:, 0], conf_bounds.iloc[:, 1],
                 color='purple', alpha=0.1, label='95% CI')
plt.xticks(forecast_range[::12], time_display[::12], rotation=45)
plt.title("AAL vs DAL Spread Projection - March 6, 2025")
plt.xlabel("Time (EST)")
plt.ylabel("Spread")
plt.legend()
plt.grid(True, linestyle='--', alpha=0.5)
plt.tight_layout()
plt.show()

# Trading summary
last_value = spread_series.iloc[-1]
forecast_avg = forecast_data.mean()
print(f"\nTrading Summary:")
print(f"Latest Spread: {last_value:.2f}")
print(f"Forecast Average: {forecast_avg:.2f}")
if forecast_avg > last_value:
    print("Action at 9:30 EST: Sell AAL, Buy DAL (spread rising)")
    print("Exit by 16:00 EST: Adjust based on spread trend")
else:
    print("Action at 9:30 EST: Buy AAL, Sell DAL (spread falling)")
    print("Exit by 16:00 EST: Adjust based on spread trend")
print(f"Final 95% CI: [{conf_bounds.iloc[-1, 0]:.2f}, {conf_bounds.iloc[-1, 1]:.2f}]")

# Forecast details
forecast_output = pd.DataFrame({
    'Time (EST)': time_display,
    'Forecast': forecast_data,
    'Lower CI': conf_bounds.iloc[:, 0],
    'Upper CI': conf_bounds.iloc[:, 1]
})
print("\nForecast for March 6, 2025 (5 hours):")
print(forecast_output.to_string(index=False))
